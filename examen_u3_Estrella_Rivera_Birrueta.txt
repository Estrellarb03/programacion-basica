#include <iostream>
#include <math.h>
using namespace std;

void caja(int num, int chocolates){
    int iterador;
    cout<<"Caja ";cout<<num;
    cout<<"\n";
    iterador = 0;
    for (int i = 0; i < 6; i++) {
        for (int j = 0; j < 8; j++) {
            iterador = iterador + 1;
            if ( chocolates >= iterador ) {
                cout<<" * ";   
            } else {
                cout<<" X ";
            }
            
        }
        cout<<"\n";
    }
}

int main()
{
    // variables
    int num_chocolates, num_cajas, mod, num_total_chocolates_por_caja = 48;
    float division;

    mod = num_chocolates % num_total_chocolates_por_caja; // modulo retorna el resido de la division

       num_cajas =  num_chocolates / num_total_chocolates_por_caja;
       
        for (int i = 1; i <= num_cajas; i++) {
            caja(i, num_total_chocolates_por_caja);cout<<"\n";
